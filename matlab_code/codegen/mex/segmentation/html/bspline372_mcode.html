1function [ref] = bspline(src, control_p, bound)
 2% bspline give 2d cubic uniform references of given source and control points
 3% Detailed explanation goes here
 4%#codegen
 5coder.inline('never')
 6
 7basic_matrix = [1,4,1,0;
 8                -3,0,3,0;
 9                3,-6,3,0;
10                -1,3,-3,1]/6;
11            
12
13top = bound(:, 1);
14left = bound(:, 2);
15x = (src(:,1) - top + 1)./30;
16y = (src(:,2) - left + 1)./30;
17tx = x - floor(x);
18ty = y - floor(y);
19
20Ax = zeros(size(tx,1), 4, 1);
21Ax(:, 1) = ones(size(tx,1), 1);
22Ax(:, 2) = tx;
23Ax(:, 3) = tx.^2;
24Ax(:, 4) = tx.^3;
25Ax = Ax*basic_matrix;
26
27Ay = zeros(size(ty,1), 4, 1);
28Ay(:, 1) = ones(size(ty,1), 1);
29Ay(:, 2) = ty;
30Ay(:, 3) = ty.^2;
31Ay(:, 4) = ty.^3;
32Ay = Ay*basic_matrix;
33
34
35point_amount = size(src, 1);
36
37[h, w] = size(control_p.x);
38
39A = zeros(point_amount, h*w, 1);
40for i=1:point_amount
41    xi = Ax(i, :);
42    yi = Ay(i, :);
43    param_matrix = zeros(h, w, 1);
44    param_matrix(1+floor(x(i)):4+floor(x(i)), 1+floor(y(i)):4+floor(y(i))) = xi'*yi;
45    param_matrix = param_matrix(:)';
46    A(i, :) = param_matrix;
47end
48
49
50ref = zeros(size(src, 1), 2, 1);
51temp1 = control_p.x;
52temp2 = control_p.y;
53
54ref(:, 1) = A*temp1(:);
55ref(:, 2) = A*temp2(:);
56
57end
