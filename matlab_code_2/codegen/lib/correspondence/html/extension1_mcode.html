1function [errors, sp_set, bspline_set] = extension(sp_set, bspline_set, Sp2, Src_lab, Ref_lab)
  2
  3%#codegen
  4coder.inline('never')
  5coder.varsize('omega');
  6coder.varsize('temp_sx');
  7coder.varsize('temp_sy');
  8coder.varsize('temp_rx');
  9coder.varsize('temp_ry');
 10
 11surface_amount = size(sp_set, 2);
 12[x, y] = size(Sp2);
 13
 14%% initialize
 15Sp_new = zeros(x, y, 1);
 16for i=1:surface_amount
 17    point_amount = size(sp_set(i).val, 1);
 18    for j=1:point_amount
 19        Sp_new(sp_set(i).val(j, 1), sp_set(i).val(j, 2)) = i;
 20    end    
 21end
 22
 23%% adjacent patch
 24patch = zeros(3, 49, 1);
 25patch(1, :) = repmat((-3:3), 1, 7);
 26patch(2, :) = sort(repmat((-3:3),1,7));
 27patch(3, :) = ones(1, 49);
 28
 29%% extension
 30errors = [];
 31
 32for i=1:surface_amount
 33    modified = true;
 34    while modified == true
 35        modified = false;
 36        superpixel = sp_set(i).val;
 37        sp_bspline = bspline_set(i);
 38        bound = min(superpixel);
 39        single_ref_set = bspline(superpixel, sp_bspline, bound);
 40        
 41        %% find adjacency area within 5 pixels
 42        omega = superpixel+repmat([0, 0], size(superpixel, 1), 1);
 43        for j= -5:5
 44            for k= abs(j)-5:5-abs(j)
 45                set = superpixel+repmat([j, k], size(superpixel, 1), 1);
 46                omega = sortrows(omega);
 47                omega = union(omega, sortrows(set), 'rows');
 48            end
 49        end
 50        omega = sortrows(omega);
 51        omega = setdiff(omega, sortrows(superpixel), 'rows');
 52        omega(omega(:,1)<1 | omega(:,2)<1 | omega(:, 1)>x | omega(:, 2)>y, :) = [];
 53        temp = omega(:, 1)+(omega(:, 2)-1)*x;
 54        omega(Sp_new(temp)~=0, :) = [];
 55        
 56        %% for all adjacent pixels ... 
 57        size_o = size(omega, 1);
 58        
 59        temp_sx = [];
 60        temp_sy = [];
 61        temp_rx = [];
 62        temp_ry = [];
 63        
 64        for j=1:size_o
 65            point_amount = size(superpixel, 1);
 66            
 67            %% find nearest pixel q from surface i
 68            temp = abs(repmat(omega(j, :), point_amount, 1) - superpixel);
 69            [val, ind] = min(temp(:, 1) + temp(:, 2));
 70            
 71            %% compute temp correspondence
 72            ref_q = single_ref_set(ind, :);
 73            ref_p = ref_q + (bspline_single(superpixel(ind, :), sp_bspline, bound, 1,0,0,1)).*(omega(j, :) - superpixel(ind,:));
 74            offset_p = ref_p - omega(j, :);
 75            
 76            %% compute patch transformation jacobian matirx
 77            [temp1] = bspline_single(superpixel(ind, :), sp_bspline, bound ,1,0,1,0);
 78            [temp2] = bspline_single(superpixel(ind, :), sp_bspline, bound ,0,1,0,1);
 79            jacobian = [temp1(1, 1), temp2(1, 1); temp1(1, 2), temp2(1, 2)];
 80            
 81            %% minimize error            
 82            error = 10000000;
 83            refp = zeros(0);
 84            
 85            for s=[0.9, 1.0, 1.1]
 86                scale = [s, 0; 0, s];
 87                for theta=[-pi/36, 0, pi/36]
 88                    rotate = [cos(theta), -sin(theta); sin(theta), cos(theta)];
 89                    jacobian_d = rotate*scale*jacobian;
 90                    for delta_x=[-1,0,1]
 91                        for delta_y=[-1,0,1]
 92                            M = [jacobian_d, offset_p'+[delta_x; delta_y]; 0,0,1];
 93                            Np = patch + repmat([omega(j,:)';0], 1, 49);
 94                            Np_ref = M*Np;
 95                            
 96                            Np = int32(Np);
 97                            Np(Np<1)=1;
 98                            Np(1, Np(1, :)>x)=x;
 99                            Np(2, Np(2, :)>y)=y;
100                            
101                            Np_ref = int32(Np_ref);
102                            Np_ref(Np_ref<1)=1;
103                            Np_ref(1, Np_ref(1, :)>x)=x;
104                            Np_ref(2, Np_ref(2, :)>y)=y;
105    
106                            Ps = getPixelsValue(Src_lab, Np');
107                            Pt = getPixelsValue(Ref_lab, Np_ref');
108                            
109                            Ps = (Ps - repmat(mean(Ps), 49, 1))./(repmat(std(Ps), 49, 1));
110                            Pt = (Pt - repmat(mean(Pt), 49, 1))./(repmat(std(Pt), 49, 1));
111
112                            
113                            t1 = sqrt(sum(Ps.^2));
114                            t2 = sqrt(sum(Pt.^2));
115                            Ps(:, 1) = Ps(:, 1)./t1(1);
116                            Ps(:, 2) = Ps(:, 2)./t1(2);
117                            Ps(:, 3) = Ps(:, 3)./t1(3);
118                            Pt(:, 1) = Pt(:, 1)./t2(1);
119                            Pt(:, 2) = Pt(:, 2)./t2(2);
120                            Pt(:, 3) = Pt(:, 3)./t2(3);
121                            
122                            e = sum(sum((Ps - Pt).^2));
123                            if e < error
124                                error = e;
125                                refp = [delta_x, delta_y];
126                            end
127                        end
128                    end
129                end
130            end
131            errors = [errors; error];
132            if error < 1
133                modified = true;
134                
135                temp_sx = [temp_sx; omega(j,1)];
136                temp_sy = [temp_sy; omega(j,2)];
137                Sp_new(omega(j,1), omega(j,2)) = i;
138                refp = refp + offset_p + omega(j, :);
139                temp_rx = [temp_rx; refp(1,1)];
140                temp_ry = [temp_ry; refp(1,2)];
141                
142                %{
143                refp = refp + offset_p + omega(j, :);
144                superpixel = [superpixel; double(omega(j, :))];
145                Sp_new(omega(j,1), omega(j,2)) = i;
146                single_ref_set = [single_ref_set; double(refp)];
147                %}
148            end
149        end
150        if modified == true
151            sp_set(i).val = [sp_set(i).val; double([temp_sx, temp_sy])];
152            single_ref_set = [single_ref_set; double([temp_rx, temp_ry])];
153            %sp_set(i).val = superpixel;
154            
155            [control_p, error] = bspline_inv(sp_set(i).val, single_ref_set);
156            item = struct('x', control_p(:,:,1), 'y', control_p(:,:,2));
157            bspline_set(i) = item;
158        end
159    end
160end
161
162end
