1function [sp_set, bspline_set, ref_set, adjacency_graph] = merging(sp_set, sp_val, bspline_set, ref_set, Sp2, Ref)
  2%#codegen
  3coder.inline('never')
  4
  5
  6%% initialize adjacency graph
  7m = max(max(Sp2));
  8adjacency_graph = zeros(m, m, 1);
  9
 10[x, y] = size(Sp2);
 11
 12%{
 13for i=1:surface_amount
 14    %area_i = find(Sp2==sp_val(i));
 15    area_i = sp_set(i).val(:,1)+(x-1)*sp_set(i).val(:,2);
 16    size_i = size(area_i, 1);
 17    for j=(i+1):surface_amount
 18        %area_j = find(Sp2==sp_val(j));
 19        area_j = sp_set(j).val(:,1)+(x-1)*sp_set(j).val(:,2);
 20        size_j = size(area_j, 1);
 21        dif = abs(repmat(area_i, 1, size_j) - repmat(area_j, 1, size_i)');
 22        if sum(find(dif == 1 | dif == x)) ~= 0
 23            adjacency_graph(i, j) = 1;
 24        end
 25    end
 26end
 27%}
 28
 29[Fx, Fy] = gradient(Sp2);
 30diffx = find(Fx~=0);
 31sx = size(diffx, 1);
 32for i=1:sx
 33    b = Sp2(diffx(i));
 34    pos = diffx(i)+x;
 35    if pos < x*y && Sp2(pos) ~= Sp2(diffx(i))
 36        a = Sp2(pos);
 37        adjacency_graph(a, b) = 1;
 38        adjacency_graph(b, a) = 1;
 39    end
 40    pos = diffx(i)-x;
 41    if pos > 0 && Sp2(pos) ~= Sp2(diffx(i))
 42        a = Sp2(pos);
 43        adjacency_graph(a, b) = 1;
 44        adjacency_graph(b, a) = 1;
 45    end
 46end
 47
 48diffy = find(Fy~=0);
 49sy = size(diffy, 1);
 50for i=1:sy
 51    b = Sp2(diffy(i));
 52    pos = diffy(i)+1;
 53    if mod(pos, x) ~= 1 && Sp2(pos) ~= Sp2(diffy(i))
 54        a = Sp2(pos);
 55        adjacency_graph(a, b) = 1;
 56        adjacency_graph(b, a) = 1;
 57    end
 58    pos = diffy(i)-1;
 59    if mod(pos, x) ~= 0 && Sp2(pos) ~= Sp2(diffy(i))
 60        a = Sp2(pos);
 61        adjacency_graph(a, b) = 1;
 62        adjacency_graph(b, a) = 1;
 63    end
 64end
 65
 66adjacency_graph = adjacency_graph(sp_val, sp_val);
 67adjacency_graph = triu(adjacency_graph, 1);
 68
 69%% merge superpixels with similiar bspline
 70Dt = [];
 71pairs = find(adjacency_graph~=0);
 72while sum(pairs) > 0
 73    surface_amount = size(sp_set, 2);
 74    i = mod(pairs(1), surface_amount);
 75    j = ceil(pairs(1)/surface_amount);
 76    union_ij = [sp_set(i).val; sp_set(j).val];
 77    ref_union = [ref_set(i).val; ref_set(j).val];
 78    [control_p, error] = bspline_inv(union_ij, ref_union);
 79    
 80    bound1 = min(union_ij, [], 1);
 81    bound2 = min(sp_set(i).val, [], 1);
 82    bound3 = min(sp_set(j).val, [], 1);
 83    
 84    item = struct('x', control_p(:,:,1), 'y', control_p(:,:,2));
 85    
 86    
 87    ref_it = (bspline(sp_set(i).val, item, bound1));
 88    ref_i = (bspline(sp_set(i).val, bspline_set(i), bound2));
 89    
 90    val_it = getPixelsValue(Ref, ref_it);
 91    val_i = getPixelsValue(Ref, ref_i);
 92    
 93    ref_jt = (bspline(sp_set(j).val, item, bound1));
 94    ref_j = (bspline(sp_set(j).val, bspline_set(j), bound3));
 95    
 96    val_jt = getPixelsValue(Ref, ref_jt);
 97    val_j = getPixelsValue(Ref, ref_j);
 98    
 99    D_sp1 = sum(sum((val_it-val_i).^2, 1), 2);
100    D_sp2 = sum(sum((val_jt-val_j).^2, 1), 2);
101    D_sp = (D_sp1+D_sp2)/size(union_ij, 1);
102    
103    
104    %D_sp = error;
105    Dt = [Dt; D_sp];
106    adjacency_graph(i, j) = 0;
107    if D_sp < 10
108        adjacency_graph(i, i:j) = adjacency_graph(i, i:j) + adjacency_graph(i:j, j)';
109        adjacency_graph(1:i, i) = adjacency_graph(1:i, i) + adjacency_graph(1:i, j);
110        adjacency_graph(i, j:end) = adjacency_graph(i, j:end) + adjacency_graph(j, j:end);
111        adjacency_graph(j, :) = [];
112        adjacency_graph(:, j) = [];
113        sp_set(i).val = union_ij;
114        sp_set(j) = [];
115        item = struct('x', control_p(:,:,1), 'y', control_p(:,:,2));
116        bspline_set(i) = item;
117        bspline_set(j) = [];
118        ref_set(i).val = ref_union;
119        %ref_set(i).val = bspline(sp_set(i).val, item, bound1);
120        ref_set(j) = [];
121    end
122    pairs = find(adjacency_graph~=0);
123end
124
125
126end
