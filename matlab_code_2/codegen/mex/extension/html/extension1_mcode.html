1function [errors, sp_set, bspline_set, Corr] = extension(sp_set, bspline_set, Sp2, Src_lab, Ref_lab)
  2
  3%#codegen
  4coder.inline('never')
  5coder.varsize('omega');
  6coder.varsize('temp_sx');
  7coder.varsize('temp_sy');
  8coder.varsize('temp_rx');
  9coder.varsize('temp_ry');
 10
 11
 12
 13surface_amount = size(sp_set, 2);
 14[x, y] = size(Sp2);
 15
 16%% initialize
 17Sp_new = zeros(x, y, 1);
 18for i=1:surface_amount
 19    point_amount = size(sp_set(i).val, 1);
 20    for j=1:point_amount
 21        Sp_new(sp_set(i).val(j, 1), sp_set(i).val(j, 2)) = i;
 22    end    
 23end
 24
 25%% adjacent patch
 26patch = zeros(3, 49, 1);
 27patch(1, :) = repmat((-3:3), 1, 7);
 28patch(2, :) = sort(repmat((-3:3),1,7));
 29patch(3, :) = ones(1, 49);
 30
 31%% extension
 32errors = [];
 33
 34for i=1:surface_amount
 35    modified = true;
 36    temp_sx = sp_set(i).val(:, 1);
 37    temp_sy = sp_set(i).val(:, 2);
 38    ref_set = bspline(sp_set(i).val, bspline_set(i), min(sp_set(i).val));
 39    
 40    while modified == true
 41        modified = false;
 42        %superpixel = sp_set(i).val;
 43        superpixel = double([temp_sx, temp_sy]);
 44        sp_bspline = bspline_set(i);
 45        bound = min(sp_set(i).val);
 46        single_ref_set = bspline(superpixel, sp_bspline, bound);
 47        
 48        %% find adjacency area within 5 pixels
 49        changed = double([temp_sx, temp_sy]);
 50        omega = changed+repmat([0, 0], size(changed, 1), 1);
 51        for j= -5:5
 52            for k= abs(j)-5:5-abs(j)
 53                set = changed+repmat([j, k], size(changed, 1), 1);
 54                omega = sortrows(omega);
 55                omega = union(omega, sortrows(set), 'rows');
 56            end
 57        end
 58        omega = sortrows(omega);
 59        omega = setdiff(omega, sortrows(changed), 'rows');
 60        omega(omega(:,1)<1 | omega(:,2)<1 | omega(:, 1)>x | omega(:, 2)>y, :) = [];
 61        temp = omega(:, 1)+(omega(:, 2)-1)*x;
 62        omega(Sp_new(temp)~=0, :) = [];
 63        
 64        %% for all adjacent pixels ... 
 65        size_o = size(omega, 1);
 66        
 67        temp_sx = [];
 68        temp_sy = [];
 69        temp_rx = [];
 70        temp_ry = [];
 71        
 72        for j=1:size_o
 73            point_amount = size(superpixel, 1);
 74            
 75            %% find nearest pixel q from surface i
 76            temp = abs(repmat(omega(j, :), point_amount, 1) - superpixel);
 77            [val, ind] = min(temp(:, 1) + temp(:, 2));
 78            
 79            %% compute patch transformation jacobian matirx
 80            [temp1] = bspline_single(superpixel(ind, :), sp_bspline, bound ,1,0,1,0);
 81            [temp2] = bspline_single(superpixel(ind, :), sp_bspline, bound ,0,1,0,1);
 82            jacobian = [temp1(1, 1), temp2(1, 1); temp1(1, 2), temp2(1, 2)];
 83            %jacobian = jacobian.*[x x; y y];
 84            
 85            %% compute temp correspondence
 86            ref_q = single_ref_set(ind, :);
 87            %ref_p = ref_q + (bspline_single(superpixel(ind, :), sp_bspline, bound, 1,0,0,1)).*(omega(j, :) - superpixel(ind,:));
 88            ref_p = ref_q + (jacobian*(omega(j, :) - superpixel(ind,:))')';
 89            if sum(ref_p < 0) || ref_p(1) > x || ref_p(2) > y
 90                continue;
 91            end
 92            offset_p = ref_p; %- omega(j, :);
 93            
 94            %% minimize error            
 95            error = 10000000;
 96            refp = zeros(0);
 97            
 98            for s=[0.9, 1.0, 1.1]
 99                scale = [s, 0; 0, s];
100                for theta=[-pi/36, 0, pi/36]
101                    rotate = [cos(theta), -sin(theta); sin(theta), cos(theta)];
102                    jacobian_d = jacobian*rotate*scale;
103                    for delta_x=[-1,0,1]
104                        for delta_y=[-1,0,1]
105                            M = [jacobian_d, offset_p'+[delta_x; delta_y]; 0,0,1];
106                            Np = patch + repmat([omega(j,:)';0], 1, 49);
107                            %Np_ref = M*Np;
108                            Np_ref = M*patch;
109    
110                            Ps = getPixelsValue(Src_lab, Np');
111                            Pt = getPixelsValue(Ref_lab, Np_ref');
112                            
113                            Ps = (Ps - repmat(mean(Ps), 49, 1))./(repmat(std(Ps), 49, 1));
114                            Pt = (Pt - repmat(mean(Pt), 49, 1))./(repmat(std(Pt), 49, 1));
115%{
116                            Ps = Ps./(repmat(std(Ps), 49, 1));
117                            Pt = Pt./(repmat(std(Pt), 49, 1));
118                            t1 = sqrt(sum(Ps.^2));
119                            t2 = sqrt(sum(Pt.^2));
120                            Ps(:, 1) = Ps(:, 1)./t1(1);
121                            Ps(:, 2) = Ps(:, 2)./t1(2);
122                            Ps(:, 3) = Ps(:, 3)./t1(3);
123                            Pt(:, 1) = Pt(:, 1)./t2(1);
124                            Pt(:, 2) = Pt(:, 2)./t2(2);
125                            Pt(:, 3) = Pt(:, 3)./t2(3);
126%}                          
127                            e = sum(sum((Ps - Pt).^2));
128                            if e < error
129                                error = e;
130                                refp = [delta_x, delta_y];
131                                %refp = Np_ref(1:2, 25)';
132                            end
133                        end
134                    end
135                end
136            end
137            if error < 150
138                modified = true;
139                
140                temp_sx = [temp_sx; omega(j,1)];
141                temp_sy = [temp_sy; omega(j,2)];
142                Sp_new(omega(j,1), omega(j,2)) = i;
143                %refp = refp + offset_p + omega(j, :);
144                %refp = (refp+offset_p)./[x y];
145                refp = (refp+offset_p);
146                temp_rx = [temp_rx; refp(1,1)];
147                temp_ry = [temp_ry; refp(1,2)];
148                
149                %{
150                refp = refp + offset_p + omega(j, :);
151                superpixel = [superpixel; double(omega(j, :))];
152                Sp_new(omega(j,1), omega(j,2)) = i;
153                single_ref_set = [single_ref_set; double(refp)];
154                %}
155            end
156        end
157        if modified == true
158            sp_set(i).val = [sp_set(i).val; double([temp_sx, temp_sy])];
159            %single_ref_set = [single_ref_set; double([temp_rx, temp_ry])];
160            single_ref_set = [];
161            ref_set = [ref_set; double([temp_rx, temp_ry])];
162            %sp_set(i).val = superpixel;
163            
164            [control_p, error] = bspline_inv(sp_set(i).val, ref_set);
165            errors = [errors; error];
166            item = struct('x', control_p(:,:,1), 'y', control_p(:,:,2));
167            bspline_set(i) = item;
168        end
169    end
170end
171
172Corr = zeros(x, y, 2);
173surface_amount = size(sp_set, 2);
174
175for i=1:surface_amount
176    point_amount = size(sp_set(i).val, 1);
177    ref_set = bspline(sp_set(i).val, bspline_set(i), min(sp_set(i).val));
178    for j=1:point_amount
179        Corr(sp_set(i).val(j, 1), sp_set(i).val(j, 2), :) = [ref_set(2), ref_set(1)];
180    end    
181end 
182
183end
