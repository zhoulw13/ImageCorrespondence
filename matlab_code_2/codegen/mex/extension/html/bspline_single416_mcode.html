1function [ref] = bspline_single(src, control_p, bound, ux, uy, vx, vy)
  2% bspline give 2d cubic uniform references of given sole point and control points
  3% Detailed explanation goes here
  4%#codegen
  5coder.inline('never')
  6%{
  7p0 = bspline(src, control_p, bound);
  8p1 = bspline(src+[1,0], control_p, bound);
  9if src(1) > bound(1)
 10    p2 = bspline(src-[1,0], control_p, bound);
 11    diffx = (p1-p2)/2;
 12else
 13    diffx = p1-p0;
 14end
 15
 16p3 = bspline(src+[0,1], control_p, bound);
 17if src(2) > bound(2)
 18    p4 = bspline(src-[0,1], control_p, bound);
 19    diffy = (p3-p4)/2;
 20else
 21    diffy = p3-p0;
 22end
 23
 24ref = zeros(size(src, 1), 2, 1);
 25if ux ~= 0
 26    ref(1) = diffx(1);
 27end
 28if vx ~= 0
 29    ref(2) = diffx(2);
 30end
 31
 32if uy ~= 0
 33    ref(1) = diffy(1);
 34end
 35if vy ~= 0
 36    ref(2) = diffy(2);
 37end
 38%}
 39
 40basic_matrix = [1,4,1,0;
 41                -3,0,3,0;
 42                3,-6,3,0;
 43                -1,3,-3,1]/6;
 44
 45top = bound(:, 1);
 46left = bound(:, 2);
 47x = (src(:,1) - top + 1)./30;
 48y = (src(:,2) - left + 1)./30;
 49tx = x - floor(x);
 50ty = y - floor(y);
 51
 52Ax = zeros(size(tx,1), 4, 1);
 53Ax(:, 1) = ones(size(tx,1), 1);
 54Ax(:, 2) = tx;
 55Ax(:, 3) = tx.^2;
 56Ax(:, 4) = tx.^3;
 57Ax = Ax*basic_matrix;
 58
 59Ax_d = zeros(size(tx,1), 4, 1);
 60if (ux+vx)~=0
 61    Ax_d(:, 1) = zeros(size(tx,1), 1);
 62    Ax_d(:, 2) = ones(size(tx,1), 1);
 63    Ax_d(:, 3) = 2*tx;
 64    Ax_d(:, 4) = 3*tx.^2;
 65    Ax_d = Ax_d*basic_matrix/30;
 66end
 67
 68Ay = zeros(size(ty,1), 4, 1);
 69Ay(:, 1) = ones(size(ty,1), 1);
 70Ay(:, 2) = ty;
 71Ay(:, 3) = ty.^2;
 72Ay(:, 4) = ty.^3;
 73Ay = Ay*basic_matrix;
 74
 75Ay_d = zeros(size(ty,1), 4, 1);
 76if (uy+vy)~=0
 77    Ay_d(:, 1) = zeros(size(ty,1), 1);
 78    Ay_d(:, 2) = ones(size(ty,1), 1);
 79    Ay_d(:, 3) = 2*ty;
 80    Ay_d(:, 4) = 3*ty.^2;
 81    Ay_d = Ay_d*basic_matrix/30;
 82end
 83
 84
 85[h, w] = size(control_p.x);
 86
 87A = zeros(1, h*w, 1);
 88Au = zeros(1, h*w, 1);
 89Av = zeros(1, h*w, 1);
 90if sum([ux, uy, vx, vy]) == 0
 91    param_matrix = zeros(h, w, 1);
 92    param_matrix(1+floor(x):4+floor(x), 1+floor(y):4+floor(y)) = Ax'*Ay;
 93    param_matrix = param_matrix(:)';
 94    A(1, :) = param_matrix;
 95    Au = A;
 96    Av = A;
 97else
 98    if ux == 0
 99        xi = Ax;
100    else
101        xi = Ax_d;
102    end
103    if uy == 0 
104        yi = Ay;
105    else
106        yi = Ay_d;
107    end
108    param_matrix = zeros(h, w, 1);
109    param_matrix(1+floor(x):4+floor(x), 1+floor(y):4+floor(y)) = xi'*yi;
110    param_matrix = param_matrix(:)';
111    Au(1, :) = param_matrix;
112
113    if vx == 0
114        xi = Ax;
115    else
116        xi = Ax_d;
117    end
118    if vy == 0
119        yi = Ay;
120    else
121        yi = Ay_d;
122    end
123    param_matrix = zeros(h, w, 1);
124    param_matrix(1+floor(x):4+floor(x), 1+floor(y):4+floor(y)) = xi'*yi;
125    param_matrix = param_matrix(:)';
126    Av(1, :) = param_matrix;
127end
128
129ref = zeros(size(src, 1), 2, 1);
130
131temp1 = control_p.x;
132temp2 = control_p.y;
133ref(:, 1) = Au*temp1(:);
134ref(:, 2) = Av*temp2(:);
135
136%{
137if (ux+uy+vx+vy) == 0
138    temp1 = control_p.x;
139    temp2 = control_p.y;
140    ref(:, 1) = Au*temp1(:);
141    ref(:, 2) = Av*temp2(:);
142else
143    temp1 = control_p.x-repmat((1:h)'-2, 1, w);
144    temp2 = control_p.y-repmat((1:w)-2, h, 1);
145    if ux ~= 0
146        ref(:,1) = Au*temp1(:)+1/30;
147    end
148    if uy ~= 0
149        ref(:,1) = Au*temp1(:);
150    end
151    if vx ~= 0
152        ref(:,2) = Av*temp2(:);
153    end
154    if vy ~= 0
155        ref(:,2) = Av*temp2(:)+1/30;
156    end
157end
158%}
159
160end
