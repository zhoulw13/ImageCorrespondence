1function [ref] = bspline_single(src, control_p, bound, ux, uy, vx, vy)
  2% bspline give 2d cubic uniform references of given sole point and control points
  3% Detailed explanation goes here
  4%#codegen
  5coder.inline('never')
  6
  7basic_matrix = [1,4,1,0;
  8                -3,0,3,0;
  9                3,-6,3,0;
 10                -1,3,-3,1]/6;
 11            
 12top = bound(:, 1);
 13left = bound(:, 2);
 14x = (src(:,1) - top + 1)./30;
 15y = (src(:,2) - left + 1)./30;
 16tx = x - floor(x);
 17ty = y - floor(y);
 18
 19Ax = zeros(size(tx,1), 4, 1);
 20Ax(:, 1) = ones(size(tx,1), 1);
 21Ax(:, 2) = tx;
 22Ax(:, 3) = tx.^2;
 23Ax(:, 4) = tx.^3;
 24Ax = Ax*basic_matrix;
 25
 26Ax_d = zeros(size(tx,1), 4, 1);
 27if (ux+vx)~=0
 28    Ax_d(:, 1) = zeros(size(tx,1), 1);
 29    Ax_d(:, 2) = ones(size(tx,1), 1);
 30    Ax_d(:, 3) = 2*tx;
 31    Ax_d(:, 4) = 3*tx.^2;
 32    Ax_d = Ax_d*basic_matrix/30;
 33end
 34
 35Ay = zeros(size(ty,1), 4, 1);
 36Ay(:, 1) = ones(size(ty,1), 1);
 37Ay(:, 2) = ty;
 38Ay(:, 3) = ty.^2;
 39Ay(:, 4) = ty.^3;
 40Ay = Ay*basic_matrix;
 41
 42Ay_d = zeros(size(ty,1), 4, 1);
 43if (uy+vy)~=0
 44    Ay_d(:, 1) = zeros(size(ty,1), 1);
 45    Ay_d(:, 2) = ones(size(ty,1), 1);
 46    Ay_d(:, 3) = 2*ty;
 47    Ay_d(:, 4) = 3*ty.^2;
 48    Ay_d = Ay_d*basic_matrix/30;
 49end
 50
 51
 52[h, w] = size(control_p.x);
 53
 54A = zeros(1, h*w, 1);
 55Au = zeros(1, h*w, 1);
 56Av = zeros(1, h*w, 1);
 57if sum([ux, uy, vx, vy]) == 0
 58    param_matrix = zeros(h, w, 1);
 59    param_matrix(1+floor(x):4+floor(x), 1+floor(y):4+floor(y)) = Ax'*Ay;
 60    param_matrix = param_matrix(:)';
 61    A(1, :) = param_matrix;
 62    Au = A;
 63    Av = A;
 64else
 65    if ux == 0
 66        xi = Ax;
 67    else
 68        xi = Ax_d;
 69    end
 70    if uy == 0 
 71        yi = Ay;
 72    else
 73        yi = Ay_d;
 74    end
 75    param_matrix = zeros(h, w, 1);
 76    param_matrix(1+floor(x):4+floor(x), 1+floor(y):4+floor(y)) = xi'*yi;
 77    param_matrix = param_matrix(:)';
 78    Au(1, :) = param_matrix;
 79
 80    if vx == 0
 81        xi = Ax;
 82    else
 83        xi = Ax_d;
 84    end
 85    if vy == 0 
 86        yi = Ay;
 87    else
 88        yi = Ay_d;
 89    end
 90    param_matrix = zeros(h, w, 1);
 91    param_matrix(1+floor(x):4+floor(x), 1+floor(y):4+floor(y)) = xi'*yi;
 92    param_matrix = param_matrix(:)';
 93    Av(1, :) = param_matrix;
 94end    
 95
 96ref = zeros(size(src, 1), 2, 1);
 97%{
 98temp1 = control_p.x;
 99temp2 = control_p.y;
100ref(:, 1) = Au*temp1(:);
101ref(:, 2) = Av*temp2(:);
102%}
103if (ux+uy+vx+vy) == 0
104    temp1 = control_p.x;
105    temp2 = control_p.y;
106    ref(:, 1) = Au*temp1(:);
107    ref(:, 2) = Av*temp2(:);
108else
109    temp1 = control_p.x-repmat((1:h)'-2, 1, w);
110    temp2 = control_p.y-repmat((1:w)-2, h, 1);
111    if ux ~= 0
112        ref(:,1) = Au*temp1(:)+1/30;
113    end
114    if uy ~= 0
115        ref(:,1) = Au*temp1(:);
116    end
117    if vx ~= 0
118        ref(:,2) = Av*temp2(:);
119    end
120    if vy ~= 0
121        ref(:,2) = Av*temp2(:)+1/30;
122    end
123end
124%}
125
126end
